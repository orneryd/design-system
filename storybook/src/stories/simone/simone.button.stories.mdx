import { Preview, Meta, Description } from '@storybook/addon-docs/blocks'
import { withKnobs, select, boolean } from '@storybook/addon-knobs'
import '@ornery/ui-button'
import '@ornery/ui-paper'
import README from '@ornery/ui-button-readme'
const label = 'Variants'
const options = {
  Primary: 'primary',
  PrimaryInverted: 'primary inverted',
  Action: 'action',
  ActionInverted: 'action inverted',
  Info: 'info',
  InfoInverted: 'info inverted',
  Success: 'success',
  SuccessInverted: 'success inverted',
  Warning: 'warning',
  WarningInverted: 'warning inverted',
  Danger: 'danger',
  DangerInverted: 'danger inverted'
}
const defaultValue = 'primary'
const groupId = 'Buttons'
const styles = {
  textTransform: 'capitalize',
  margin: '1rem'
}

<Meta
  title="Simone/Buttons"
  decorators={[withKnobs]}
  parameters={{
    component: 'ui-button',
    options: { 
      selectedPanel: 'storybookjs/knobs/panel'
     }
  }}
/>

<Description>
    {README}
</Description>

<Preview withToolbar>
  <Story name="All">
    <>
      <link rel="stylesheet" type="text/css" href="/simone/overrides.css" />
      {Object.values(options).map(variant => {
        return (<div style={{margin: '10px 0'}}>
              <ui-button variant={variant}>
                <span class="btn condensed">{variant}</span>
              </ui-button>
            </div>)
      })}
    </>
  </Story>
</Preview>

<Story name="Button">
  <>
  <link rel="stylesheet" type="text/css" href="/simone/overrides.css" />
  <ui-button
    style={styles}
    variant={select(label, options, defaultValue, groupId)}
    disabled={boolean('Disabled', false, groupId)}
  >
    <span class="btn condensed">{select(label, options, defaultValue, groupId)}</span>
  </ui-button>
  </>
</Story>

<Story name="Anchor">
  <>
  <link rel="stylesheet" type="text/css" href="/simone/overrides.css" />
  <ui-button
    style={styles}
    href="https://www.google.com"
    target="_blank"
    variant={select(label, options, 'outlined rounded', groupId)}
    disabled={boolean('Disabled', false, groupId)}
  >
    <span class="btn condensed">external link</span>
  </ui-button>
  </>
</Story>
